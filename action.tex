%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\subsection*{Operators}
\textit{Use like: \codein{[\#]<op><mot>} where <mot> are the simple cursor motions described above. Eg, \codein{y2w} yanks 2 word; \codein{3dd} deletes 3 lines; \codein{cf(} changes to first open paren.}\\
\entry{25mm}{\#y}{\underline{y}ank \# chars}\\
\entry{25mm}{x\quad X}{delete aft / bef cursor}\\
\entry{25mm}{r\quad R}{\ul{r}eplace char(s) @ cursor}\\
\entry{25mm}{c\quad C}{\ul{c}hange at cursor}\\
\entry{25mm}{p\quad P}{paste bef. / aft.}\\
\entry{25mm}{yy\quad dd}{\ul{yy}ank \ul{dd}elete line(s)}\\
\entry{25mm}{cf<char>}{\ul{c}hange thru found <char>}\\
\entry{25mm}{>>\quad <<}{shift current line in, out}\\
\entry{25mm}{==}{zero indent, current line}\\
\entry{25mm}{\#=}{autoindent \# lines}\\
\entry{25mm}{\textasciitilde}{change case}\\
\entry{25mm}{g~~}{change case, whole line}\\
\entry{25mm}{gu\quad gU}{$\Delta$ case to \ul{U}pper, lower (\ul{u})}\\
\entry{25mm}{<c-A>|<c-X>}{inc (\underline{a}dd) / dec (\underline{x}) numb}\\
