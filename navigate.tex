\subsection*{Marks}
\textit{Mark a place in the file. Buffer-specific marks $\in$ [a-z]; globals $\in$ [A-Z].}\\
\entry{25mm}{m<letter>}{mark $\to$ letter}\\
\entry{25mm}{`<letter>}{jump to mark}\\
\entry{25mm}{'<letter>}{jump to mark's line}\\
\entry{25mm}{:marks}{list all marks}\\
\entry{25mm}{:marks ab}{list marks a, b}\\
\entry{25mm}{:delmarks ab}{delete marks a, b}\\
\entry{30mm}{d`<letter>\quad d'<letter>}{delete $\to$ mark}\\
\entry{30mm}{c`<letter>\quad c'<letter>}{change, here to ...}\\
\entry{30mm}{y`<letter>\quad y'<letter>}{yank, here to ...}\\
\entry{30mm}{[`<am>\quad']<am>}{jump $\to$ \say{auto mark}}\\

\textit{``Automarks'' include cursor location:}\\
\begin{itemize}
    \item \begin{code}'\end{code} ... before last jump
    \item \begin{code}]\end{code} ... before last inserted text
    \item \begin{code}[\end{code} ... end of last inserted text
    \item \begin{code}\textquotedbl\end{code} ... last cursor when left file
    \item \hlc[pastelgrey]{`<}, \hlc[pastelgrey]{`>} ... last visual select start, stop
\end{itemize}
