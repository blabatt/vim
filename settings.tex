\subsection*{Saving}
\textit{.viminfo saves marks, registers, vars, options \codein{:set viminfo=<opt>*} where <opt> $\in$:}\\
\entry{25mm}{\textquotesingle <\#>}{max \# files to save marks }\\
\entry{25mm}{f}{save globals }\\
\entry{25mm}{r<loc>}{save removeable media }\\
\entry{25mm}{h}{:nohlsearch }\\
\entry{25mm}{\textbackslash\textquotedbl<\#>}{\# lines saved per buff. }\\
\entry{25mm}{\%}{restore buff. list }\\
\entry{25mm}{@\#}{size, input-line history }\\
